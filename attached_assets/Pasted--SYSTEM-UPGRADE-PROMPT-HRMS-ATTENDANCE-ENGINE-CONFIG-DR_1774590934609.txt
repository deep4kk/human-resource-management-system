## 🔄 ATTENDANCE ENGINE (IMPORTANT)

Refactor attendance processing into a pipeline:

```js
function processAttendance(attendance, company_id) {
  const settings = getAttendanceSettings(company_id);
  const flags = getFeatureFlags(company_id);

  if (settings.shift_rule.enabled) applyShiftLogic(attendance, settings);
  if (settings.late_rule.enabled) applyLateLogic(attendance, settings);
  if (settings.working_hours_rule.enabled) applyWorkingHours(attendance, settings);

  if (flags.geo_fencing && settings.geo_fencing.enabled) {
    validateGeoLocation(attendance);
  }

  if (flags.face_recognition && settings.face_recognition.enabled) {
    verifyFace(attendance);
  }

  if (settings.auto_regularization.enabled) {
    handleRegularization(attendance, settings);
  }

  triggerNotifications(attendance, settings, flags);

  return attendance;
}
```

## ⏱️ LATE RULE ENGINE

```js
function applyLateLogic(attendance, settings) {
  const lateAfter = settings.late_rule.late_after_time;
  const grace = settings.late_rule.grace_minutes;

  if (attendance.checkInTime > addMinutes(lateAfter, grace)) {
    attendance.isLate = true;
    attendance.lateCount += 1;
  }

  if (attendance.lateCount > settings.late_rule.allowed_lates) {
    applyPenalty(attendance, settings.late_rule);
  }
}
```

## ⏳ WORKING HOURS LOGIC

```js
function applyWorkingHours(attendance, settings) {
  const hours = calculateWorkHours(attendance);

  if (hours < settings.working_hours_rule.absent_threshold) {
    attendance.status = "Absent";
  } else if (hours < settings.working_hours_rule.half_day_threshold) {
    attendance.status = "Half Day";
  } else {
    attendance.status = "Present";
  }
}
```
